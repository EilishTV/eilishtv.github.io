        <div class="maint-bg"></div>

        <div style="z-index: 2; width: 90%; max-width: 420px;">
            <div id="countdown">
                <div class="time-box">
                    <span id="days">00</span>
                    <label>Días</label>
                </div>

                <div class="time-box">
                    <span id="hours">00</span>
                    <label>Hrs</label>
                </div>

                <div class="time-box">
                    <span id="minutes">00</span>
                    <label>Min</label>
                </div>

                <div class="time-box">
                    <span id="seconds">00</span>
                    <label>Seg</label>
                </div>
            </div>

            <p style="font-size: 0.8rem; margin-bottom: 10px; letter-spacing: 1px;">
                ACCESS CODE
            </p>

            <div class="code-inputs">
                <input type="text" maxlength="1" inputmode="numeric" autocomplete="one-time-code" class="code-box">
                <input type="text" maxlength="1" inputmode="numeric" autocomplete="one-time-code" class="code-box">
                <input type="text" maxlength="1" inputmode="numeric" autocomplete="one-time-code" class="code-box">
                <input type="text" maxlength="1" inputmode="numeric" autocomplete="one-time-code" class="code-box">
            </div>

            <div class="loading-message" id="loading-message"></div>
            <div class="error-message" id="error-message">
                Ocurrió un error al cargar. Reintentando...
            </div>
        </div>
    </div>

     <script>
        (() => {
            'use strict';

            const SECRET_CODE = '1201';

            const overlay = document.getElementById('maintenance-overlay');
            const content = document.getElementById('original-content');
            const inputs = document.querySelectorAll('.code-box');
            const errorMessage = document.getElementById('error-message');
            const loadingMessage = document.getElementById('loading-message');

            let unlocked = false;
            let retryCount = 0;
            const maxRetries = 3;

            function safeShow(element, display = 'block') {
                if (element) {
                    element.style.display = display;
                }
            }

            function safeHide(element) {
                if (element) {
                    element.style.display = 'none';
                }
            }

            function unlock() {
                if (unlocked) return;

                unlocked = true;

                safeHide(overlay);
                safeShow(content, 'block');

                document.body.style.overflowX = 'hidden';
                document.body.style.overflowY = 'auto';

                loadOriginalScript();
            }

            function loadOriginalScript() {
                loadingMessage.textContent = 'Loading...';

                const existingScript = document.querySelector('script[data-identify="true"]');
                if (existingScript) return;

                const script = document.createElement('script');
                script.src = '../js/identify.js';
                script.defer = true;
                script.dataset.identify = 'true';

                script.onload = () => {
                    loadingMessage.textContent = '';
                    safeHide(errorMessage);
                    console.log('identify.js cargado correctamente');
                };

                script.onerror = () => {
                    retryCount++;
                    console.error('Error cargando identify.js');

                    if (retryCount <= maxRetries) {
                        loadingMessage.textContent = `Retrying... (${retryCount}/${maxRetries})`;

                        setTimeout(() => {
                            loadOriginalScript();
                        }, 1500);
                    } else {
                        loadingMessage.textContent = '';
                        safeShow(errorMessage);
                    }
                };

                document.body.appendChild(script);
            }

            function checkCode() {
                let currentCode = '';

                inputs.forEach(input => {
                    currentCode += input.value;
                });

                if (currentCode.length === SECRET_CODE.length && currentCode === SECRET_CODE) {
                    unlock();
                }
            }

            inputs.forEach((input, index) => {

                input.addEventListener('input', (e) => {
                    let value = e.target.value.replace(/[^0-9]/g, '');

                    if (value.length > 1) {
                        value = value.slice(-1);
                    }

                    input.value = value;

                    if (value && index < inputs.length - 1) {
                        inputs[index + 1].focus();
                    }

                    checkCode();
                });

                input.addEventListener('keydown', (e) => {
                    if (e.key === 'Backspace' && !input.value && index > 0) {
                        inputs[index - 1].focus();
                    }
                });

                input.addEventListener('paste', (e) => {
                    e.preventDefault();

                    const pastedData = (e.clipboardData || window.clipboardData)
                        .getData('text')
                        .replace(/\D/g, '')
                        .slice(0, inputs.length);

                    pastedData.split('').forEach((char, i) => {
                        if (inputs[i]) {
                            inputs[i].value = char;
                        }
                    });

                    checkCode();
                });
            });

            function updateTimer() {
                try {
                    const now = new Date();

                    const target = new Date(now);
                    target.setDate(now.getDate() + ((6 - now.getDay() + 7) % 7 || 7));
                    target.setHours(0, 0, 0, 0);

                    const diff = target.getTime() - now.getTime();

                    if (diff <= 0) {
                        unlock();
                        return;
                    }

                    const days = Math.floor(diff / 86400000);
                    const hours = Math.floor((diff % 86400000) / 3600000);
                    const minutes = Math.floor((diff % 3600000) / 60000);
                    const seconds = Math.floor((diff % 60000) / 1000);

                    document.getElementById('days').textContent = String(days).padStart(2, '0');
                    document.getElementById('hours').textContent = String(hours).padStart(2, '0');
                    document.getElementById('minutes').textContent = String(minutes).padStart(2, '0');
                    document.getElementById('seconds').textContent = String(seconds).padStart(2, '0');
                } catch (error) {
                    console.error('Error en el timer:', error);
                }
            }

            setInterval(updateTimer, 1000);
            updateTimer();

            window.addEventListener('error', (event) => {
                console.error('Error global:', event.error);
            });

            window.addEventListener('unhandledrejection', (event) => {
                console.error('Promise rechazada:', event.reason);
            });

            document.addEventListener('visibilitychange', () => {
                if (!document.hidden) {
                    updateTimer();
                }
            });

        })();
    </script>
